program fibonacci;

function int fib(int base) {
    if (base == 0) then {
        return(0);
    }
    if (base == 1) then {
        return(1);
    }
    return(fib(base - 1) + fib(base - 2));
}

main()
    var int: first, second, fibN, i, temp;
{
    fibN = 8;
    write(fib(fibN));

    first = 0;
    second = 1;
    temp = 1;
    from i = 0 to fibN do {
        first = second;
        second = temp;
        temp = first + second;
    }
    write(first);
}
