program matMutl;

main ()
    var int: mat1[2][3], mat2[3][2], res[2][2];
        int: i, j, x;
{
    mat1[0][0] = 3;
    mat1[0][1] = -2;
    mat1[0][2] = 5;
    mat1[1][0] = 3;
    mat1[1][1] = 0;
    mat1[1][2] = 4;

    mat2[0][0] = 2;
    mat2[0][1] = 3;
    mat2[1][0] = -9;
    mat2[1][1] = 0;
    mat2[2][0] = 0;
    mat2[2][1] = 4;

    from i = 0 to 2 do {
        from j = 0 to 2 do {
            res[i][j] = 0;
            from x = 0 to 3 do {
                res[i][j] = res[i][j] + mat1[i][x] * mat2[x][j];
            }
        }
    }

    from i = 0 to 2 do {
        from j = 0 to 2 do {
            write(res[i][j]);
        }
    }
}
