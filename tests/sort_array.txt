program sort;
var int: arr[6];

main()
    var int: i, j, min, temp;
{
    arr[0] = 16;
    arr[1] = 12;
    arr[2] = 8;
    arr[3] = 9;
    arr[4] = 1;
    arr[5] = 9;

    from i = 0 to 5 do {
        min = i;
        from j = i + 1 to 6 do {
            if (arr[j] < arr[min]) then {
                min = j;
            }
        }
        temp = arr[min];
        arr[min] = arr[i];
        arr[i] = temp;
    }

    from i = 0 to 6 do {
        write(arr[i]);
    }
}
