program pruebaStat;

function float test(float temporal) {
    return(temporal + 10.0);
    write(1 + 1);
}

main()
   var dataFrame: data;
        int: rows, cols;
        float: dataMean, dataMode, dataVar;
{
    load(data, "./tests/test_input_data.txt", cols, rows);
    mean(data, 1, dataMean);
    write(dataMean);
    mode(data, 0, dataMode);
    write(dataMode);
    variance(data, 0, dataVar);
    write(dataVar);

    write(test(dataVar));

    covariance(data, 1, dataVar);
    write("COVARIANCE",dataVar);

    graphBar(data, 1);
    scatter(data, 1, 2);
}
